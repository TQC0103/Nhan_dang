\begin{frame}{Trong cascade: một stage được tạo như thế nào?}
  \small
  \begin{enumerate}
    \setlength{\itemsep}{6pt}
    \item \textbf{Sinh feature:} liệt kê nhiều Haar-like / MB-LBP tại nhiều vị trí--kích thước trên window \(24\times24\) (tính nhanh nhờ \textit{integral image}).\ \cite{viola2001rapid,viola2004robust}
    \item \textbf{Chọn weak learner (vòng \(m\)):} với mỗi feature, tìm ngưỡng \(\tau\) (và polarity) để tối thiểu \textbf{weighted error}; lấy feature tốt nhất làm \(h_m(x)\).
    \item \textbf{Boosting \(\rightarrow\) strong classifier:} lặp \(m=1..M\), cộng dồn \(\alpha_m h_m(x)\) để được \(H(x)=\sum_{m=1}^{M}\alpha_m h_m(x)\).
    \item \textbf{Chỉnh threshold stage \(\theta_i\):} đặt \(\theta_i\) để đạt \textbf{recall} mục tiêu \(d_i\) (chấp nhận FP tạm thời), rồi đưa sang stage tiếp theo.\ \cite{viola2004robust}
  \end{enumerate}
  \vspace{3pt}
  {\scriptsize Ghi chú: AsymBoost chỉ \textit{tăng phạt} khi bỏ sót positive để đẩy recall cao hơn trong cascade.\ \cite{wu2008asymmetric}}
\end{frame}

\begin{frame}{AdaBoost: trọng số và cập nhật trọng số}
  \small
  \begin{block}{Đại lượng chính}
    Trọng số mẫu \(w_i\) được chuẩn hóa \(\sum_i w_i = 1\); weak classifier \(h_m(x)\in\{-1,+1\}\); nhãn \(y_i\in\{-1,+1\}\).
  \end{block}
  \begin{block}{Cập nhật (dạng chuẩn)}
    \[
      \epsilon_m=\sum_i w_i\,[y_i\ne h_m(x_i)],\qquad
      \alpha_m=\tfrac{1}{2}\ln\frac{1-\epsilon_m}{\epsilon_m}
    \]
    \[
      w_i \leftarrow w_i\exp\!\big(-\alpha_m y_i h_m(x_i)\big)\quad\text{(sau đó chuẩn hóa lại)}.
    \]
  \end{block}
  \begin{itemize}
    \setlength{\itemsep}{6pt}
    \item Mẫu bị phân loại sai \(\Rightarrow\) \(y_i h_m(x_i)=-1\) \(\Rightarrow\) \(w_i\) tăng \(\Rightarrow\) vòng sau tập trung vào \textbf{mẫu khó}.
    \item Trong cascade, cơ chế này giúp stage đầu học các feature rẻ nhưng phân biệt tốt để loại âm thật nhanh.\ \cite{viola2001rapid,viola2004robust}
  \end{itemize}
\end{frame}

\begin{frame}{AsymBoost: vì sao cần học bất đối xứng?}
  \begin{itemize}
    \setlength{\itemsep}{7pt}
    \item \textbf{Mất cân bằng mạnh:} số window nền \(\gg\) số window chứa mặt \(\Rightarrow\) dễ thiên về loại nhầm mặt (FN).
    \item \textbf{Chi phí FN cao:} bỏ sót mặt là lỗi nặng hơn so với báo động giả tạm thời (FP) ở các stage đầu.
    \item \textbf{Ý tưởng:} cho phép FP tăng \textit{có kiểm soát} để đẩy \textbf{recall} lên rất cao; cascade nhiều tầng sẽ lọc FP dần.\ \cite{li2012facechapter,wu2008asymmetric}
  \end{itemize}
\end{frame}

\begin{frame}{AsymBoost: khác AdaBoost ở bước cập nhật trọng số}
  \begin{block}{Quy tắc cập nhật}
    Giữ nguyên \(\epsilon_m\), \(\alpha_m\) như AdaBoost; chỉ sửa \textbf{weight update} với hệ số phạt \(k>1\) cho \textbf{positive bị phân loại sai}:
    \[
      w_i^{(m+1)} = w_i^{(m)}\exp\!\big(-\alpha_m y_i h_m(x_i)\big) \cdot \begin{cases}
        k & \text{nếu } y_i = +1 \text{ và } h_m(x_i) = -1 \\
        1 & \text{trường hợp còn lại}
      \end{cases}
    \]
  \end{block}
  {\small \(k\) làm tăng trọng số các mẫu mặt bị bỏ sót \(\Rightarrow\) boosting ưu tiên kéo chúng về đúng ở vòng sau.\ \cite{li2012facechapter,wu2008asymmetric}}
\end{frame}

\begin{frame}{AsymBoost trong cascade: tác động thực tế}
  \begin{itemize}
    \setlength{\itemsep}{7pt}
    \item \textbf{Stage đầu:} ưu tiên \textbf{không bỏ sót} \(\Rightarrow\) có thể giữ lại nhiều candidate (FP tăng).
    \item \textbf{Stage sau:} dùng nhiều weak learner hơn + hard negative mining để triệt FP còn lại.
    \item \textbf{Hệ quả:} AsymBoost giúp đạt điểm làm việc thiên về \textbf{recall} (phù hợp nhu cầu cascade face detector).\ \cite{wu2008asymmetric}
  \end{itemize}
\end{frame}

\begin{frame}{Insight từ minh họa cài đặt (kết nối sang phương pháp hiện đại)}
  \begin{itemize}
    \setlength{\itemsep}{8pt}
    \item \textbf{Hyperparam nhạy:} \texttt{minNeighbors} quyết định trade-off recall vs false positives.\ \cite{li2012facechapter}
    \item \textbf{Hạn chế cấu trúc:} cửa sổ cố định + Haar-like khó nắm bắt biến thiên tư thế và tỉ lệ rất nhỏ \(\Rightarrow\) recall thấp trên WIDER FACE.\ \cite{yang2016widerface}
    \item \textbf{Động lực cho SCRFD:} thay vì tăng thủ công cascade/đặc trưng, hướng hiện đại tối ưu phân bổ compute và phân phối mẫu theo tỉ lệ để cải thiện mặt nhỏ trong ngân sách tính toán.\ \cite{guo2022scrfd}
  \end{itemize}

\section{Chương II -- Phương pháp bài báo SCRFD}

\begin{frame}{(2) Phương pháp chính của bài báo: SCRFD}
  \begin{block}{Mục tiêu}
    Tối ưu \textbf{độ chính xác--chi phí tính toán} cho phát hiện khuôn mặt, đặc biệt là \textbf{khuôn mặt nhỏ} trong ảnh độ phân giải thấp (ví dụ VGA).\ \cite{guo2022scrfd}
  \end{block}
  \begin{itemize}
    \setlength{\itemsep}{4pt}
    \item \textbf{Computation Redistribution (CR):} tái phân bổ FLOPs giữa \textit{backbone--neck--head}.
    \item \textbf{Sample Redistribution (SR):} tái phân bổ mẫu dương theo thang tỉ lệ qua tìm kiếm tập scale augmentation.
    \item \textbf{Triển khai:} random search trong search space thiết kế gọn \(\Rightarrow\) tạo ``họ'' SCRFD theo nhiều mức GFLOPs.
  \end{itemize}
\end{frame}

\begin{frame}{Kiến trúc tổng quát (Backbone--Neck--Head)}
  \begin{columns}[T,onlytextwidth]
    \column{0.52\textwidth}
    \begin{itemize}
      \setlength{\itemsep}{5pt}
      \item Nền tảng theo hướng single-shot, có neck đa tỉ lệ và head dự đoán dày đặc.\ \cite{guo2022scrfd}
      \item Điểm nhấn của SCRFD không phải ``module mới'', mà là \textbf{phân bổ đúng tài nguyên} cho đúng nơi cần.
    \end{itemize}

\begin{frame}{Computation Redistribution (CR): ý tưởng và tham số}

    \item Kết luận thực nghiệm: ưu tiên FLOPs cho backbone (đặc biệt tầng nông) và nén neck/head trong nhiều chế độ compute.

\begin{frame}{Sample Redistribution (SR): tái phân bổ mẫu theo tỉ lệ}
  \begin{itemize}
    \setlength{\itemsep}{5pt}
    \item Quan sát: phát hiện mặt nhỏ phụ thuộc mạnh vào \textbf{phân phối tỉ lệ} và augmentation.\ \cite{guo2022scrfd}
    \item Thiết kế zoom-in / zoom-out search-able:
      \begin{itemize}
        \setlength{\itemsep}{2pt}
        \item chọn tập scale rời rạc \(S=\{s_{\min}, s_{\min}+0.1, \ldots, s_{\max}\}\),
        \item random search nhiều tập scale \(\Rightarrow\) chọn tập có AP tốt nhất.
      \end{itemize}
    \item Ý nghĩa: tăng xác suất xuất hiện khuôn mặt nhỏ ở các feature map cần thiết.
  \end{itemize}
\end{frame}

\begin{frame}{Sample Redistribution (SR): Tối ưu hóa khớp Anchor}
  \begin{itemize}
    \item \textbf{Vấn đề:} Các tham số tăng cường quy mô truyền thống thường được thiết kế thủ công và cố định cho mọi mạng, dẫn đến việc khớp anchor không tối ưu.
    \item \textbf{Giải pháp SR:} Thực hiện tìm kiếm ngẫu nhiên trên 320 tập scale augmentation để tìm ra phân phối mẫu tốt nhất cho từng kiến trúc cụ thể.
    \item \textbf{Hiệu quả thực nghiệm (trên SCRFD-2.5GF):}
      \begin{itemize}
        \item Tập scale tối ưu tìm được:
        \begin{equation}
          \mathbb{S} = \{0.5, 0.7, 0.8, 1.0, 1.1, 1.2, 1.4, 1.5, 1.8, 2.0, 2.3, 2.6\}
        \end{equation}
        \item Tăng đáng kể số lượng Positive Anchors tại các thang đo nhỏ như 16 và 32.
        \item AP trên tập Hard tăng thêm 0.48\% so với việc mở rộng tập scale thủ công.
      \end{itemize}
  \end{itemize}
\end{frame}

\section{Liên hệ sách--paper và đánh giá}

\begin{frame}{(3) Mối liên hệ giữa chương sách và SCRFD}
  \begin{itemize}
    \setlength{\itemsep}{5pt}
    \item \textbf{Đa tỉ lệ:} tháp ảnh (Viola--Jones) \(\rightarrow\) pyramid feature (CNN/neck) và scale augmentation (SR).\ \cite{li2012facechapter,guo2022scrfd}
    \item \textbf{Tiết kiệm tính toán:} cascade loại sớm theo mức độ khó \(\rightarrow\) CR tái phân bổ FLOPs đúng tầng/cấu phần quan trọng.
    \item \textbf{Khung 3 bước giữ nguyên:} sinh ứng viên / chấm điểm / hợp nhất (NMS) vẫn là xương sống triển khai.
  \end{itemize}
\end{frame}

\begin{frame}{Khác biệt cốt lõi: cổ điển vs nghiên cứu hiện đại}
  \small
  \renewcommand{\arraystretch}{1.15}
  \begin{tabular}{@{}p{0.32\textwidth}p{0.32\textwidth}p{0.32\textwidth}@{}}

    \midrule
    Đặc trưng & Haar/MB-LBP thủ công & CNN đa tầng học sâu \\
    Tối ưu compute & Cascade loại sớm & CR (FLOPs backbone/neck/head) \\
    Xử lý tỉ lệ & Image pyramid + cửa sổ trượt & FPN/PAFPN + SR (scale search) \\
    Thiết kế hệ & Rule-based + boosting & Data-driven + random/NAS search \\
    \bottomrule

\begin{frame}{(4) Đánh giá kỹ thuật: Điểm mạnh cốt lõi}
  \begin{itemize}
    \setlength{\itemsep}{8pt}
    \item \textbf{Phương pháp luận thực tiễn:} Khẳng định rõ thông điệp không áp dụng thụ động các cấu trúc mạng phân loại tổng quát sang bài toán đặc thù, mà chủ động tái thiết kế không gian mạng để giữ chi tiết cho khuôn mặt nhỏ.
    \item \textbf{Đa dạng ngân sách tính toán:} Tạo ra một dải tham số phong phú từ 0.5GFLOPs tới 34GFLOPs, đáp ứng mọi giới hạn hiệu năng của phần cứng.\ \cite{guo2022scrfd}
    \item \textbf{Tính nhất quán của quá trình phát triển:} Phối hợp nhịp nhàng giữa hai trục tài nguyên và dữ liệu mà không làm cồng kềnh mô hình, giúp duy trì tư duy xuyên suốt từ nghiên cứu quy mô nhỏ tới đám đông phức tạp.
  \end{itemize}
\end{frame}

\begin{frame}{(4) Đánh giá kỹ thuật: Hạn chế và rào cản kỹ thuật}
  \begin{itemize}
    \setlength{\itemsep}{8pt}
    \item \textbf{Chi phí NAS khổng lồ:} Quá trình tìm kiếm kiến trúc thông qua Empirical Bootstrap cần huấn luyện hàng trăm mô hình trên cụm GPU, tạo rào cản tái lập rất lớn cho các nhóm nghiên cứu quy mô nhỏ.\ \cite{guo2022scrfd}
    \item \textbf{Các ràng buộc giới hạn việc tìm kiếm:} Không gian được đóng khung bằng các ngưỡng chặn thủ công và dùng kiến trúc khối cơ bản mô-đun hóa định sẵn, chưa vươn tới việc tìm kiếm các cấu trúc vi mô bên trong từng khối tích chập.
    \item \textbf{Yếu điểm miền dữ liệu và đối tượng siêu nhỏ:} Việc áp dụng thực tiễn gặp thách thức trước sự lệch miền so với dữ liệu WIDER FACE; cùng với đó các khuôn mặt dưới ngưỡng neo vẫn bị giảm độ nhạy do không đủ đặc trưng.
  \end{itemize}
\end{frame}

\begin{frame}{(4) Đánh giá kỹ thuật: Phạm vi ứng dụng công nghiệp}
  \begin{itemize}
    \setlength{\itemsep}{8pt}
    \item \textbf{Triển khai ở lớp thiết bị biên:} Các biến thể cấu hình nhẹ có thể chạy trực tiếp, độc lập với độ trễ thấp trên smartphone, hệ thống thu phát IoT hay các camera an ninh nhận diện tại cửa.
    \item \textbf{Triển khai ở lớp máy chủ:} Thông qua những phiên bản phức tạp hơn, hệ thống vẫn duy trì sự nhạy bén và độ vạch đích cao với các khung hình góc rộng trong sân vận động hay ga tàu, nơi con người trông hết sức nhỏ bé.
    \item \textbf{Hướng dẫn tinh chỉnh kiến trúc vĩ mô:} Bài học từ SCRFD khuyến khích người làm công nghiệp không chạy lại NAS từ con số âm, mà dùng luôn nguyên lý của CR/SR để cân chỉnh tham số theo miền đích mới.
  \end{itemize}
\end{frame}

\begin{frame}{(5) Nhận định cá nhân về bài báo SCRFD}
  \begin{itemize}
    \setlength{\itemsep}{6pt}
    \item \textbf{Bước tiến về tư duy thiết kế:} Nghiên cứu này chứng minh giải pháp ``tối ưu đúng chỗ'' đem lại kết quả vượt trội và kinh tế hơn việc tạo ra một cấu trúc mạng mới phức tạp.
    \item \textbf{Ưu tiên tính thực tiễn đại trà:} Việc thu hẹp không gian tìm kiếm kết hợp thống kê Empirical Bootstrap cung cấp quy trình tối ưu dễ tái hiện, mở ra tiềm năng tạo các detector nhẹ nhưng hiệu quả trên thiết bị IoT/di động.
    \item \textbf{Phát huy triệt để dữ liệu cục bộ:} Cơ chế Sample Redistribution thông qua scale augmentation là minh chứng cho việc tận dụng tốt thuộc tính phân bố đặc thù của khuôn mặt nhỏ trực tiếp trong quá trình tính loss.
  \end{itemize}
\end{frame}

\begin{frame}{Hướng mở rộng khả thi và ứng dụng}
  \begin{columns}[c,onlytextwidth]

      \item \textbf{Tích hợp Knowledge Distillation:} Có thể dùng mô hình SCRFD khổng lồ làm Teacher, huấn luyện bản cực nhỏ làm Student để duy trì độ chính xác cao nhưng chi phí tính toán siêu thấp.
