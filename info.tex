\section*{Thông tin đề tài}
\phantomsection
\addcontentsline{toc}{section}{Thông tin đề tài}

\begin{itemize}
  \item \textbf{Chủ đề:} Sự tiến hóa của kỹ thuật phát hiện khuôn mặt, từ phương pháp cổ điển đến hướng tối ưu kiến trúc và dữ liệu trong học sâu.
  \item \textbf{Chương sách được chọn:} \textit{Handbook of Face Recognition}, ấn bản thứ hai, Chapter 11 \textit{Face Detection}, trang 277--301, tác giả Stan Z. Li và Jianxin Wu.
  \item \textbf{Bài báo phân tích:} \textit{Sample and Computation Redistribution for Efficient Face Detection}, ICLR 2022, tác giả Jia Guo, Jiankang Deng, Alexandros Lattas, Stefanos Zafeiriou.
  \item \textbf{Mục tiêu báo cáo giữa kỳ:} nắm khung phát hiện ba bước, tái lập detector Viola--Jones theo ví dụ trong chương sách và phân tích phương pháp SCRFD theo bài báo gốc.
  \item \textbf{Phạm vi:} không tái tạo pha tìm kiếm NAS do giới hạn tài nguyên tính toán; phần đối chiếu SCRFD dựa trên số liệu công bố trong bài báo; phần minh họa triển khai mở rộng sẽ thực hiện ở báo cáo cuối kỳ.
\end{itemize}
